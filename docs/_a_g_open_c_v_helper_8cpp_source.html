53 void AGOpenCVHelper::calculateLinearFunctionCoeffsUsingPoints(const cv::Point &pointOne, const cv::Point &pointTwo, std::vector<double> &coeffs)
54 {
55  if (pointTwo.x == pointOne.x) {
56  coeffs.push_back(0.00);
57  coeffs.push_back(0.00);
58  return;
59  }
60  double aCoeff = (double)(pointTwo.y - pointOne.y) / (double)(pointTwo.x - pointOne.x);
61  double bCoeff = (double)(pointOne.y) - aCoeff * (double)(pointOne.x);
62  coeffs.push_back(aCoeff);
63  coeffs.push_back(bCoeff);
64 }

153 void AGOpenCVHelper::linkTwoImagesTogether(const cv::Mat &imageOne, const cv::Mat &imageTwo, cv::Mat &outputImage, const ImageDirection &imageDirection, AGError &error)
154 {
155  if (!imageOne.data || !imageTwo.data) {
156  error = { true, "linkTwoImagesTogether: imageOne or imageTwo has no data." }; return;
157  }
158 
159  int totalCols = 0;
160  int totalRows = 0;
161 
162  if (imageDirection == Up || imageDirection == Down) {
163  totalCols = (imageTwo.cols > imageOne.cols) ? imageTwo.cols : imageOne.cols;
164  totalRows = imageOne.rows + imageTwo.rows;
165  }
166  else {
167  totalCols = imageOne.cols + imageTwo.cols;
168  totalRows = (imageTwo.rows > imageOne.rows) ? imageTwo.rows : imageOne.rows;
169  }
170 
171  Mat totalImage(Size(totalCols, totalRows), imageOne.type());
172 
173  switch (imageDirection) {
174  case Up:
175  imageTwo.copyTo(totalImage(Rect(0, 0, imageTwo.cols, imageTwo.rows)));
176  imageOne.copyTo(totalImage(Rect(0, imageTwo.rows, imageTwo.cols, imageTwo.rows)));
177  break;
178 
179  case Down:
180  imageOne.copyTo(totalImage(Rect(0, 0, imageOne.cols, imageOne.rows)));
181  imageTwo.copyTo(totalImage(Rect(0, imageOne.rows, imageTwo.cols, imageTwo.rows)));
182  break;
183 
184  case Left:
185  imageTwo.copyTo(totalImage(Rect(0, 0, imageOne.cols, imageOne.rows)));
186  imageOne.copyTo(totalImage(Rect(imageOne.cols, 0, imageTwo.cols, imageTwo.rows)));
187  break;
188 
189  case Right:
190  imageOne.copyTo(totalImage(Rect(0, 0, imageOne.cols, imageOne.rows)));
191  imageTwo.copyTo(totalImage(Rect(imageOne.cols, 0, imageTwo.cols, imageTwo.rows)));
192  break;
193  }
194 
195  outputImage = totalImage;
196 }

198 void AGOpenCVHelper::linkTwoImagesTogetherAndDrawMatches(const AGImage &imageOne, const AGImage &imageTwo, const std::vector<cv::DMatch> &matches, cv::Mat &outputImage, const ImageDirection &imageDirection, AGError &error)
199 {
200  if (!imageOne.image.data || !imageTwo.image.data) {
201  error = { true, "linkTwoImagesTogetherAndDrawMatches: imageOne or imageTwo has no data." }; return;
202  }
203 
204  if (imageOne.keypoints.empty() || imageTwo.keypoints.empty()) {
205  error = { true, "linkTwoImagesTogetherAndDrawMatches: imageOne.keypoints or imageTwo.keypoints are empty." }; return;
206  }
207 
208  Mat first = imageOne.image.clone();
209  Mat second = imageTwo.image.clone();
210 
211  if (imageOne.image.type() == CV_8UC1 && imageTwo.image.type() == CV_8UC1) {
212  AGOpenCVHelper::convertImageToColor(first);
213  AGOpenCVHelper::convertImageToColor(second);
214  }
215  else if (imageOne.image.type() == CV_8UC4 && imageTwo.image.type() == CV_8UC4) {
216  cvtColor(first, first, CV_RGBA2BGR);
217  cvtColor(second, second, CV_RGBA2BGR);
218  }
219 
220  for (int i = 0; i < imageOne.keypoints.size(); i++) {
221  circle(first, imageOne.keypoints[i].pt, 3, Scalar(255, 0, 0));
222  }
223  for (int i = 0; i < imageTwo.keypoints.size(); i++) {
224  circle(second, imageTwo.keypoints[i].pt, 3, Scalar(0, 255, 0));
225  }
226 
227  AGError checkError = {false, ""};
228  AGOpenCVHelper::linkTwoImagesTogether(first, second, outputImage, imageDirection, checkError);
229  if (checkError.isError) {
230  error = { true, "linkTwoImagesTogetherAndDrawMatches: drawMatches method failed inside with error: " + checkError.description }; return;
231  }
232 
233  switch (imageDirection) {
234  case Up:
235  for (int i = 0; i < matches.size(); i++) {
236  Point pointTwo = imageTwo.keypoints[matches[i].trainIdx].pt;
237  Point pointOne(imageOne.keypoints[matches[i].queryIdx].pt.x, imageOne.keypoints[matches[i].queryIdx].pt.y + imageTwo.image.rows);
238  line(outputImage, pointOne, pointTwo, Scalar(0, 0, 255));
239  }
240  break;
241 
242  case Down:
243  for (int i = 0; i < matches.size(); i++) {
244  Point pointOne = imageOne.keypoints[matches[i].queryIdx].pt;
245  Point pointTwo(imageTwo.keypoints[matches[i].trainIdx].pt.x, imageTwo.keypoints[matches[i].trainIdx].pt.y + imageOne.image.rows);
246  line(outputImage, pointOne, pointTwo, Scalar(0, 0, 255));
247  }
248  break;
249 
250  case Left:
251  for (int i = 0; i < matches.size(); i++) {
252  Point pointTwo = imageTwo.keypoints[matches[i].trainIdx].pt;
253  Point pointOne(imageOne.keypoints[matches[i].queryIdx].pt.x + imageTwo.image.cols, imageOne.keypoints[matches[i].queryIdx].pt.y);
254  line(outputImage, pointOne, pointTwo, Scalar(0, 0, 255));
255  }
256  break;
257 
258  case Right:
259  for (int i = 0; i < matches.size(); i++) {
260  Point pointOne = imageOne.keypoints[matches[i].queryIdx].pt;
261  Point pointTwo(imageTwo.keypoints[matches[i].trainIdx].pt.x + imageOne.image.cols, imageTwo.keypoints[matches[i].trainIdx].pt.y);
262  line(outputImage, pointOne, pointTwo, Scalar(255, 0, 0));
263  }
264  break;
265  }
