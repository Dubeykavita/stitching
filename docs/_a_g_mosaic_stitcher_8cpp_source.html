76 void AGMosaicStitcher::performStitching(vector<vector<AGImage>> &imagesMatrix, Mat &outputImage)
77 {
78  int imageWidth = imagesMatrix.front().front().width;
79  int imageHeight = imagesMatrix.front().front().height;
80 
81  int midXCoor = floor((imagesMatrix.size() - 1) * 0.5);
82  int midYCoor = floor((imagesMatrix.front().size() - 1) * 0.5);
83 
84  int offset = 0;
85  int outputImageWidth = imageWidth * (int)imagesMatrix.size() + 2 * offset;
86  int outputImageHeigth = imageHeight * (int)imagesMatrix.front().size() + 2 * offset;
87 
88  outputImage = Mat::zeros(outputImageHeigth, outputImageWidth, CV_8UC4);
89 
90  this->xShift = midXCoor * imageWidth + offset;
91  this->yShift = midYCoor * imageHeight + offset;
92 
93  ImageDirection imageDirection;
94  if (imagesMatrix.front().size() > imagesMatrix.size()) {
95  for (int x = 0; x < midXCoor; x++) {
96  imageDirection = Right;
97  this->applyStitchingAlgorithm(imagesMatrix[x][midYCoor], imagesMatrix[x + 1][midYCoor],
98  imageDirection, this->transformsMatrix[x][midYCoor]);
99  }
100 
101  for (int x = midXCoor + 1; x < imagesMatrix.size(); x++) {
102  imageDirection = Left;
103  this->applyStitchingAlgorithm(imagesMatrix[x][midYCoor], imagesMatrix[x - 1][midYCoor],
104  imageDirection, this->transformsMatrix[x][midYCoor]);
105  }
106 
107  for (int x = 0; x < imagesMatrix.size(); x++) {
108  for (int y = midYCoor - 1; y >= 0; y--) {
109  imageDirection = Down;
110  this->applyStitchingAlgorithm(imagesMatrix[x][y], imagesMatrix[x][y + 1],
111  imageDirection, this->transformsMatrix[x][y]);
112  }
113  for (int y = midYCoor + 1; y < imagesMatrix.front().size(); y++) {
114  imageDirection = Up;
115  this->applyStitchingAlgorithm(imagesMatrix[x][y], imagesMatrix[x][y - 1],
116  imageDirection, this->transformsMatrix[x][y]);
117  }
118  }
119  } else {
120  for (int y = 0; y < midYCoor; y++) {
121  imageDirection = Down;
122  this->applyStitchingAlgorithm(imagesMatrix[midXCoor][y], imagesMatrix[midXCoor][y + 1],
123  imageDirection, this->transformsMatrix[midXCoor][y]);
124  }
125 
126  for (int y = midYCoor + 1; y < imagesMatrix.front().size(); y++) {
127  imageDirection = Up;
128  this->applyStitchingAlgorithm(imagesMatrix[midXCoor][y], imagesMatrix[midXCoor][y - 1],
129  imageDirection, this->transformsMatrix[midXCoor][y]);
130  }
131 
132  for (int y = 0; y < imagesMatrix.front().size(); y++) {
133  for (int x = midXCoor - 1; x >= 0; x--) {
134  imageDirection = Right;
135  this->applyStitchingAlgorithm(imagesMatrix[x][y], imagesMatrix[x + 1][y],
136  imageDirection, this->transformsMatrix[x][y]);
137  }
138  for (int x = midXCoor + 1; x < imagesMatrix.size(); x++) {
139  imageDirection = Left;
140  this->applyStitchingAlgorithm(imagesMatrix[x][y], imagesMatrix[x - 1][y],
141  imageDirection, this->transformsMatrix[x][y]);
142  }
143  }
144  }
145 
146  // All images need to be shifted to the place where reference image is located
147  Mat baseShiftTransform;
148  AGOpenCVHelper::createShiftMatrix(baseShiftTransform, this->xShift, this->yShift);
149  for (int x = 0; x < imagesMatrix.size(); x++) {
150  for (int y = 0; y < imagesMatrix.front().size(); y++) {
151  cvtColor(imagesMatrix[x][y].image, imagesMatrix[x][y].image, CV_GRAY2BGRA);
152  warpAffine(imagesMatrix[x][y].image, imagesMatrix[x][y].image, baseShiftTransform, outputImage.size());
153  warpAffine(imagesMatrix[x][y].mask, imagesMatrix[x][y].mask, baseShiftTransform, outputImage.size());
154  }
155  }
156 
157  // Displaying rest of the images
158 
159  for (int x = 0; x < imagesMatrix.size(); x++) {
160  for (int y = 0; y < imagesMatrix.front().size(); y++) {
161  if (x != midXCoor || y != midYCoor) {
162  Mat transformedImage = imagesMatrix[x][y].image;
163  Mat mask = imagesMatrix[x][y].mask;
164  if (imagesMatrix.front().size() > imagesMatrix.size()) {
165  if (y > midYCoor) {
166  for (int yTrans = y; yTrans > midYCoor; yTrans--) {
167  warpAffine(transformedImage, transformedImage, transformsMatrix[x][yTrans],
168  outputImage.size());
169  warpAffine(mask, mask, transformsMatrix[x][yTrans], outputImage.size());
170  }
171  } else if (y < midYCoor) {
172  for (int yTrans = y; yTrans < midYCoor; yTrans++) {
173  warpAffine(transformedImage, transformedImage, transformsMatrix[x][yTrans],
174  outputImage.size());
175  warpAffine(mask, mask, transformsMatrix[x][yTrans], outputImage.size());
176  }
177  }
178 
179  if (x > midXCoor) {
180  for (int xTrans = x; xTrans > midXCoor; xTrans--) {
181  warpAffine(transformedImage, transformedImage, transformsMatrix[xTrans][midYCoor],
182  outputImage.size());
183  warpAffine(mask, mask, transformsMatrix[xTrans][midYCoor], outputImage.size());
184  }
185  } else if (x < midXCoor) {
186  for (int xTrans = x; xTrans < midXCoor; xTrans++) {
187  warpAffine(transformedImage, transformedImage, transformsMatrix[xTrans][midYCoor],
188  outputImage.size());
189  warpAffine(mask, mask, transformsMatrix[xTrans][midYCoor], outputImage.size());
190  }
191  }
192  } else {
193  if (x > midXCoor) {
194  for (int xTrans = x; xTrans > midXCoor; xTrans--) {
195  warpAffine(transformedImage, transformedImage, transformsMatrix[xTrans][y],
196  outputImage.size());
197  warpAffine(mask, mask, transformsMatrix[xTrans][y], outputImage.size());
198  }
199  } else if (x < midXCoor) {
200  for (int xTrans = x; xTrans < midXCoor; xTrans++) {
201  warpAffine(transformedImage, transformedImage, transformsMatrix[xTrans][y],
202  outputImage.size());
203  warpAffine(mask, mask, transformsMatrix[xTrans][y], outputImage.size());
204  }
205  }
206 
207  if (y > midYCoor) {
208  for (int yTrans = y; yTrans > midYCoor; yTrans--) {
209  warpAffine(transformedImage, transformedImage, transformsMatrix[midXCoor][yTrans],
210  outputImage.size());
211  warpAffine(mask, mask, transformsMatrix[midXCoor][yTrans], outputImage.size());
212  }
213  } else if (y < midYCoor) {
214  for (int yTrans = y; yTrans < midYCoor; yTrans++) {
215  warpAffine(transformedImage, transformedImage, transformsMatrix[midXCoor][yTrans],
216  outputImage.size());
217  warpAffine(mask, mask, transformsMatrix[midXCoor][yTrans], outputImage.size());
218  }
219  }
220  }
221 // transformedImage.copyTo(outputImage, mask);
222 // addWeighted(transformedImage, 1.0, outputImage, 1.0, 0.0, outputImage);
223  }
224  }
225  }
226  // Displaying reference (unchanged) image on the final grid
227 // imagesMatrix[midXCoor][midYCoor].image.copyTo(outputImage, imagesMatrix[midXCoor][midYCoor].mask);
228 // addWeighted(imagesMatrix[midXCoor][midYCoor].image, 1.0, outputImage, 1.0, 0.0, outputImage);
229 
230  vector<AGImage> imagesToBlend;
231  for (auto &imageRow : imagesMatrix) {
232  for (auto &image : imageRow) {
233  imagesToBlend.push_back(image);
234  }
235  }
236 
237  AGError error;
238  AGImageBlender::blendImages(imagesToBlend, outputImage, error);
239  if (error.isError) {
240  cout << error.description << endl;
241  }

414 void AGMosaicStitcher::findMatchesWithBruteForce(vector<ImageFeatures> &imagesFeatures, MatchesInfo &matchesInfo)
415 {
416  /*
417  * Brute Force Matcher is called twice, because he gives different results that depends on which image
418  * is passed as train, and which as query. Probably this is because he iterates through all keypoints from query
419  * image and makes 1-many relationship to keypoints from train image (we want many-many relationships). As a
420  * solution we combine the result from both calls to bfMatcher and later in algorithm we are removing duplicates.
421  */
422 
423  BFMatcher bfMatcherOne;
424  MatchesInfo matchesInfoOne;
425  bfMatcherOne.match(imagesFeatures[0].descriptors, imagesFeatures[1].descriptors, matchesInfoOne.matches);
426 
427  BFMatcher bfMatcherTwo;
428  MatchesInfo matchesInfoTwo;
429  bfMatcherTwo.match(imagesFeatures[1].descriptors, imagesFeatures[0].descriptors, matchesInfoTwo.matches);
430 
431  for (int i = 0; i < matchesInfoOne.matches.size(); i++) {
432  DMatch match;
433  match.queryIdx = matchesInfoOne.matches[i].queryIdx;
434  match.trainIdx = matchesInfoOne.matches[i].trainIdx;
435  match.distance = matchesInfoOne.matches[i].distance;
436  matchesInfo.matches.push_back(match);
437  }
438 
439  for (int i = 0; i < matchesInfoTwo.matches.size(); i++) {
440  DMatch match;
441  match.queryIdx = matchesInfoTwo.matches[i].trainIdx;
442  match.trainIdx = matchesInfoTwo.matches[i].queryIdx;
443  match.distance = matchesInfoTwo.matches[i].distance;
444  matchesInfo.matches.push_back(match);
445  }
446 }

529 // Calculates angle between matches and x axis, then calculates the median angle of all matches, and deletes that ones that have angle greater than angleParameter
530 void AGMosaicStitcher::filterMatchesBasedOnSlopeAndLength(AGImage &imageOne,
531  AGImage &imageTwo,
532  vector<DMatch> &matches,
533  vector<DMatch> &filtredMatches,
534  ImageDirection imageDirection)
535 {
536  if (matches.empty() || imageOne.keypoints.empty() || imageTwo.keypoints.empty()) {
537  return;
538  }
539  vector<double> angles;
540 // vector<double> lengths;
541  vector<pair<int, double>> matchIndexAnglePair;
542  for (int i = 0; i < matches.size(); i++) {
543  Point pointOne = imageOne.keypoints[matches[i].queryIdx].pt;
544  Point pointTwo = imageTwo.keypoints[matches[i].trainIdx].pt;
545  if (imageDirection == Up || imageDirection == Down) {
546  pointTwo.y += imageOne.height;
547  }
548  else {
549  pointTwo.x += imageOne.width;
550  }
551  pointTwo.y *= -1;
552  pointOne.y *= -1;
553 
554  vector<double> coeff;
555  AGOpenCVHelper::linearFunctionCoeffsUsingPoints(pointOne, pointTwo, coeff);
556 // lengths.push_back(AGOpenCVHelper::distanceBetweenPoints(pointOne, pointTwo));
557  double angle = atan(coeff[A_COEFF]) * 180 / M_PI;
558  if (angle < 0) {
559  angle += 180;
560  }
561  else if (angle == 0) {
562  angle = 90;
563  }
564  angles.push_back(angle);
565  }
566 
567  if (this->parameters.clustering) {
568  vector<double> lengthsCluster;
569  vector<double> anglesCluster;
570 // this->clusterArrayWithinRange(lengths, lengthsCluster, this->parameters.lengthParameter);
571  this->clusterArrayWithinRange(angles, anglesCluster, this->parameters.angleParameter);

625 // Filtering matches based on coordinates x or y. The main idea is that images cannot be shifted too far from each other. You can control the allowed shift by error paramter (expressed in percentage)
626 void AGMosaicStitcher::filterMatchesBasedOnPlacement(AGImage &imageOne,
627  AGImage &imageTwo,
628  vector<DMatch> &matches,
629  vector<DMatch> &filtredMatches,
630  ImageDirection imageDirection)
631 {
632  if (matches.empty() || !imageOne.image.data || !imageTwo.image.data
633  || this->parameters.shiftParameter > 1.0 || this->parameters.shiftParameter < 0.0) {
634  return;
635  }
636  for (int i = 0; i < matches.size(); i++) {
637  Point pointOne = imageOne.keypoints[matches[i].queryIdx].pt;
638  Point pointTwo = imageTwo.keypoints[matches[i].trainIdx].pt;
639  bool inRange = false;
640  if (imageDirection == Up || imageDirection == Down) {
641  if (abs(pointOne.x - pointTwo.x) < imageOne.width * this->parameters.shiftParameter) {
642  inRange = true;
643  }
644  }
645  else {
646  if (abs(pointOne.y - pointTwo.y) < imageOne.height * this->parameters.shiftParameter) {
647  inRange = true;
648  }
649  }
650  if (inRange) {
651  double distance = 0.0;
652  switch (imageDirection) {
653  case Up:
654  distance = sqrt(pow(abs(pointTwo.x - pointOne.x), 2.0) +
655  pow(abs((pointOne.y + imageTwo.height) -pointTwo.y), 2.0));
656  break;
657 
658  case Down:
659  distance = sqrt(pow(abs(pointTwo.x - pointOne.x), 2.0) +
660  pow(abs((pointTwo.y + imageOne.height) - pointOne.y), 2.0));
661  break;
662 
663  case Left:
664  distance = sqrt(pow(abs((pointOne.x + imageTwo.width) - pointTwo.x), 2.0) +
665  pow(abs(pointTwo.y - pointOne.y), 2.0));
666  break;

670  pow(abs(pointTwo.y - pointOne.y), 2.0));
