156 void AGImageLoader::loadTilesInMosaicNumber(std::vector<std::vector<AGImage>> &tiles, int mosaicNumber, AGError &error)
157 {
158  // This method should always give images in left-right coordinate system
159  // Right now images in folder are in left-bottom coordinate system
160  // Additionally images need to be rotated (180 degrees) which is flipping horizontal then vertical
161 
162  if (mosaicNumber < 0) {
163  error = { true, "loadTilesInMosaicNumber: Wrong mosaic number passed." }; return;
164  }
165 
166  unsigned long xCoordinatePosition = this->tilesBaseName.find("_X");
167  xCoordinatePosition += 2;
168 
169  for (int y = 0; ; ++y) {
170  Mat image = imread(this->getTilePathAtPosition(0, y, mosaicNumber), CV_LOAD_IMAGE_GRAYSCALE);
171  if (!image.data) {
172  break;
173  }
174  else {
175  vector<AGImage> nextColumn;
176  tiles.push_back(nextColumn);
177  }
178  }
179 
180  for (int y = (int)tiles.size() - 1; y >= 0; --y) {
181  for (int x = 0; ; ++x) {
182  Mat image = imread(this->getTilePathAtPosition(x, y, mosaicNumber), CV_LOAD_IMAGE_GRAYSCALE);
183  if (!image.data) {
184  break;
185  }
186  else {
187  AGOpenCVHelper::rotateImage(image, 180);
188  AGImage imageInfo(image, x, (int)tiles.size() - y - 1, image.cols, image.rows, this->getTileNameAtPosition(x, y));
189  tiles[x].push_back(imageInfo);
190  }
191  }
192  }
193  if (tiles.empty()) {
194  error = { true, "loadTilesInMosaicNumber: There is no images to load. Check path and image name in configuration file." }; return;
195  }
196 }

198 string AGImageLoader::getTilePathAtPosition(int x, int y, int mosaicNumber)
199 {
200  if (x < 0 || y < 0) {
201  return "";

203 
204  string tilePath = this->parameters.mosaicsDirectoryAbsolutePath + "/" + "mosaic_" + to_string(mosaicNumber);
205 
206  tilePath += "/";
207  tilePath += this->getTileNameAtPosition(x, y);
208 
209  return tilePath;

212 string AGImageLoader::getTileNameAtPosition(int x, int y)
213 {
214  if (x < 0 || y < 0) {
215  return "";
216  }
217 
218  string tileName = this->tilesBaseName;
219 
220  unsigned long xCoordinatePosition = this->tilesBaseName.find("_X");
221  xCoordinatePosition += 2;
222 
223  tileName.erase(xCoordinatePosition, 1);
224  tileName.insert(xCoordinatePosition, to_string(x));
225 
226  unsigned long yCoordinatePosition = tileName.find("_Y");
227  yCoordinatePosition += 2;
228 
229  tileName.erase(yCoordinatePosition, 1);
230  tileName.insert(yCoordinatePosition, to_string(y));
231 
232  return tileName;
233 }
