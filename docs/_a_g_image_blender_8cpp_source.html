22 void AGImageBlender::blendImages(std::vector<AGImage> &images, cv::Mat &outputImage, AGError &error)
23 {
24  AGOpenCVHelper::createEmptyMatWithSize(outputImage, images.front().image.size(), CV_8U);
25 
26  for (auto &image : images) {
27  AGError checkError;
28  AGImageBlender::calculateDistanceTransformOfImage(image, checkError);
29  if (checkError.isError) {
30  error = { true, "blendImages: Error while calculating distance transform. " + checkError.description }; return;
31  }
32  }
33 
34  for (int row = 0; row < outputImage.rows; ++row) {
35  for (int col = 0; col < outputImage.cols; ++col) {
36  Point currentPoint = Point(col, row);
37  int numeratorSum = 0, denominatorSum = 0;
38  for (auto &image : images) {
39  AGError checkError;
40  bool isPixelWhite = AGOpenCVHelper::isPixelWhiteInImage(image.mask, currentPoint, checkError);
41  if (checkError.isError) {
42  error = { true, "blendImages: Error while checking white pixel. " + checkError.description }; return;
43  }
44  if (isPixelWhite) {
45  AGError checkErrorDT, checkErrorI;
46  int weight = AGOpenCVHelper::getPixelValueAtPointInImage(image.distanceTransform, currentPoint, checkErrorDT);
47  if (checkErrorDT.isError) {
48  error = { true, "blendImages: Error while getting pixel value of distance transform. " + checkError.description }; return;
49  }
50  numeratorSum += weight * AGOpenCVHelper::getPixelValueAtPointInImage(image.image, currentPoint, checkErrorI);
51  if (checkErrorI.isError) {
52  error = { true, "blendImages: Error while getting pixel value of image. " + checkErrorI.description }; return;
53  }
54  denominatorSum += weight;
55  }
56  }
57  int pixelValue = 0;
58  if (denominatorSum != 0) {
59  pixelValue = numeratorSum / denominatorSum;
60  }
61  AGError checkError;
62  AGOpenCVHelper::setPixelValueAtPointInImage(outputImage, currentPoint, pixelValue, checkError);
63  if (checkError.isError) {
64  error = { true, "blendImages: Error while setting pixel value. " + checkError.description }; return;
65  }
66  }
67  }
68 }
